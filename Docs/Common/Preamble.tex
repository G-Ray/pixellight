% When using "book" as document class, there are gigantic borders wasting a lot of space (especially source code examples are no fun this way).
% - We can't use the "a4wide"-package because it's depreciated and may cause problems, see e.g. ftp://ftp.dante.de/tex-archive/info/l2tabu/english/l2tabuen.pdf
% - Configurating the layout like
%     \usepackage{geometry}
%     \geometry{a4paper, top=25mm, left=30mm, right=10mm, bottom=30mm, headsep=10mm, footskip=12mm}
%   would of course work, but hacking around in standard layouts isn't really a good idea
% - Komascript-Book (http://ftp.uni-erlangen.de/mirrors/CTAN/macros/latex/contrib/koma-script/scrguien.pdf) looks good, so we use it
\documentclass[a4paper,12pt]{scrbook}

% Packages
\usepackage[T1]{fontenc}
\usepackage[latin1]{inputenc}
\usepackage{color}
\usepackage{verbatim}
\usepackage{fancybox}
\usepackage[alsoload=binary, mode=text,loctolang={DE:german},decimalsymbol=comma]{siunitx}	% Units (bit, byte and so on)
\usepackage[dvipdfm]{graphicx}
\usepackage{textcomp}
\flushbottom

% Avoid writing over linebreak (can happen with technical quotes, but it's not useful when we print outside the visible page)
\tolerance 1414
\hbadness 1414
\emergencystretch 1.5em
\hfuzz 0.3pt
\widowpenalty=10000
\clubpenalty=10000
\vfuzz \hfuzz
\raggedbottom

% Links
\usepackage[dvipdfm, colorlinks=true, urlcolor=black, linkcolor=black]{hyperref}
\urlstyle{same} % Use normal font for Urls
\usepackage[all]{hypcap} % Link to the image, not the image caption

% Table
\usepackage{tabularx}
\usepackage{threeparttablex}
\usepackage{booktabs}
\usepackage{ltxtable}

% Code listening
\usepackage{color,listings}
\definecolor{darkgreen}{rgb}{0,0.5,0}
\definecolor{lightgray}{gray}{0.95}
\lstset{language=C++,
captionpos=b,
frame=single,
basicstyle=\ttfamily,
keywordstyle=\color{blue},
commentstyle=\color{darkgreen},
stringstyle=\color{red},
backgroundcolor=\color{lightgray},
numbers=left,
numberstyle=\tiny,
numbersep=5pt,
breaklines=true,
showstringspaces=false,
columns=fullflexible,	% Please, don't change e.g. "export ANDROID_SDK=~/android-sdk-linux_x86" into "export ANDROID_SDK =~/ android -sdk - linux_x86"
keepspaces=true,		% Please, keep spaces, don't change our source code automatically
tabsize=2,				% The page width is limited, so just replace a tab with two spaces instead of four (as used in the PixelLight source codes)
emph={double,bool,int,unsigned,char,true,false,void},
emphstyle=\color{blue},
emph={Assert,Test},
emphstyle=\color{red},
emph={[2]\using,\#define,\#ifdef,\#endif}, emphstyle={[2]\color{blue}}
}
